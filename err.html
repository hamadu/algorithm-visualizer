Computes Error Array: O(N)

        public static int[] errorFunction(char[] a) {
            int len = a.length;
            int[] err = new int[len+1];
            err[0] = -1;
            for (int i = 2 ; i <= len ; i++) {
                int now = err[i-1];
                while (now > 0 && a[i-1] != a[now]) {
                    now = err[now];
                }
                if (a[i-1] == a[now]) {
                    now++;
                } else {
                    now = 0;
                }
                err[i] = now;
            }
            return err;
        }
